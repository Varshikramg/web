Tasks Controller (important part)
class TasksController < ApplicationController
  before_action :set_column

  def create
    @task = @column.tasks.build(task_params)
    if @task.save
      redirect_to board_path(@column.board)
    else
      render :new
    end
  end

  def update
    @task = @column.tasks.find(params[:id])
    if @task.update(task_params)
      redirect_to board_path(@column.board)
    else
      render :edit
    end
  end

  def destroy
    @task = @column.tasks.find(params[:id])
    @task.destroy
    redirect_to board_path(@column.board)
  end

  private

  def set_column
    @column = Column.find(params[:column_id])
  end

  def task_params
    params.require(:task).permit(:title, :description, :column_id)
  end
end

class ConversionsController < ApplicationController
  before_action :set_conversion, only: [:show]

  def index
    render json: Conversion.all
  end

  def show
    render json: @conversion
  end

  def create
    @conversion = Conversion.new(conversion_params)

    if valid_currency?(@conversion.from_currency) && valid_currency?(@conversion.to_currency)
      rate = get_rate(@conversion.from_currency, @conversion.to_currency)
      @conversion.result = @conversion.amount * rate

      if @conversion.save
        render json: @conversion, status: :created
      else
        render json: { errors: @conversion.errors }, status: :unprocessable_entity
      end
    else
      render json: { error: "Invalid currency" }, status: :bad_request
    end
  end

  private

  def set_conversion
    @conversion = Conversion.find(params[:id])
  end

  def conversion_params
    params.require(:conversion).permit(:amount, :from_currency, :to_currency)
  end

  # Simple static rates (for exam)
  def get_rate(from, to)
    rates = {
      "USD_INR" => 83,
      "INR_USD" => 0.012,
      "USD_EUR" => 0.9,
      "EUR_USD" => 1.1
    }

    rates["#{from}_#{to}"] || 1
  end

  def valid_currency?(currency)
    ["USD", "INR", "EUR"].include?(currency)
  end
end

class StudentsController < ApplicationController
  before_action :set_student, only: [:show, :update, :destroy]

  # GET /students
  def index
    render json: Student.all
  end

  # GET /students/:id
  def show
    render json: @student
  end

  # POST /students
  def create
    student = Student.new(student_params)

    if student.save
      render json: student, status: :created
    else
      render json: { errors: student.errors.full_messages }, status: :unprocessable_entity
    end
  end

  # PUT /students/:id
  def update
    if @student.update(student_params)
      render json: @student
    else
      render json: { errors: @student.errors.full_messages }, status: :unprocessable_entity
    end
  end

  # DELETE /students/:id
  def destroy
    @student.destroy
    render json: { message: "Student deleted successfully" }
  end

  # 🔍 SEARCH BY COURSE
  def search
    if params[:course].present?
      students = Student.where(course: params[:course])
      render json: students
    else
      render json: { error: "Course parameter missing" }, status: :bad_request
    end
  end

  private

  def set_student
    @student = Student.find_by(id: params[:id])
    unless @student
      render json: { error: "Student not found" }, status: :not_found
    end
  end

  def student_params
    params.require(:student).permit(:name, :email, :course)
  end
end

class Vehicle
  attr_accessor :brand, :speed

  def initialize(brand, speed)
    @brand = brand
    @speed = speed
  end

    puts "Speed: #{@speed} km/h"
  end

# Subclass: Car
class Car < Vehicle
  attr_accessor :seats

  def initialize(brand, speed, seats)
    super(brand, speed)
    @seats = seats
  end

  # Method overriding
  def display_info
    super
    puts "Seats: #{@seats}"
    puts "Type: Car"
  end
end

# Subclass: Bike

  attr_accessor :engine_capacity

  def initialize(brand, speed, engine_capacity)
    super(brand, speed)
    @engine_capacity = engine_capacity
  end

  def display_info
    super

    puts "Type: Bike"
  end

  end

# Library class
class Library
  def initialize
    @books = []
  end

  # Add book
  def add_book(title, author)
    @books << Book.new(title, author)
    puts "Book added successfully."
  end

  # View books
  def view_books
    puts "----- Library Books -----"

    if @books.empty?
      puts "No books available."
      return
    end

    @books.each_with_index do |book, index|
      status = book.issued ? "Issued" : "Available"
      puts "#{index + 1}. #{book.title} by #{book.author} - #{status}"
    end
  end

  # Issue book
  def issue_book(title)
    book = @books.find { |b| b.title == title }

    if book.nil?
      puts "Book not found."
    elsif book.issued
      puts "Book already issued."
    else
      book.issued = true
      puts "Book issued successfully."
    end
  end

  # Return book
  def return_book(title)
    book = @books.find { |b| b.title == title }

    if book.nil?
      puts "Book not found."
    elsif !book.issued
      puts "Book was not issued."
    else
      book.issued = false
      puts "Book returned successfully."
    end
  end
end

class ConvertersController < ApplicationController
  def create
    converter = Converter.new(converter_params)

    # Perform conversion
    case converter.conversion_type
    when "meter_to_km"
      converter.result = converter.value / 1000.0
    when "gram_to_kg"
      converter.result = converter.value / 1000.0
    else
      render json: { error: "Invalid conversion type" }, status: :bad_request
      return
    end

    if converter.save
      render json: converter, status: :created
    else
      render json: { errors: converter.errors.full_messages }, status: :unprocessable_entity
    end
  end

  def index
    render json: Converter.all
  end

  def show
    converter = Converter.find_by(id: params[:id])

    if converter
      render json: converter
    else
      render json: { error: "Record not found" }, status: :not_found
    end
  end

  private

  def converter_params
    params.require(:converter).permit(:value, :conversion_type)
  end
end
